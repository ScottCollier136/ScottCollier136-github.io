Find Duplicate Numbers

This function takes in an integer array, and prints out what numbers are duplicated, and at what indexes.


                //The odd spacing of the"<>"'s is due to html not liking them next to a character, and falsely reading a new element
                public static void FindDuplicateNumbers(int[] array) {

                    int i;

                    HashMap< Integer, ArrayList< Integer >> myHash = new HashMap< Integer, ArrayList< Integer >>();


                    for(i=0; i< array.length; i++){
                        if(myHash.containsKey(array[i])) {
                            myHash.get(array[i]).add(i);
                        }
                        else{
                            myHash.put(array[i],new ArrayList< Integer >());
                            myHash.get(array[i]).add(i);
                        }
                    }

                    for(Map.Entry< Integer, ArrayList< Integer >> mapElement : myHash.entrySet() ){
                        if(mapElement.getValue().size() > 1){
                            System.out.println("The value " + mapElement.getKey() + " was duplicated at indexes " + mapElement.getValue());
                        }
                    }


                }
